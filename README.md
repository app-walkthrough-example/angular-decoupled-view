        var deferred = $q.defer();

        $http({
          method: 'get',
          // setting the herders Authorization to the to getToken()
          // so we are able to pass the token for Authorization
          headers: {
            Authorization: 'Bearer ' + loginService.getToken()

          deferred.reject(err);
